```C++
void IOStatusRowStopCheck() {
  IOSTATUS* tmp;
  for (int i = 0; i < IO_Status_Row.size(); i++) {
    tmp = (IOSTATUS*)IO_Status_Row.get(i);
    if (!(IsInEdgeRow(tmp->id, tmp->board) && tmp->type == 1)) {
      tmp->check = 0;
      tmp->rising_edge = 0;
      tmp->falling_edge = 0;
    }
  }
}
```
